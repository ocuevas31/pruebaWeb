int dev()

int sig(int n)
{
	
	return n+1;
}

char fun(char x[])
{
	return x[0];
}


char funfun(char x[],int p)
{
	return x[p];
}


void mos(int w[],int tam)
{
	int ind;
	
	printf("\n|");
	
	for(ind=0;ind<tam;ind++)
	printf("%d ",w[ind]);
	printf("|\n");
	
	
}

int funA(int p)
{
	
	return p+p;
}
int funB(int x)
{
	return x*x;
}

int mifun(int k)
{
	int local;
	
	printf("Dentro k=%d ",k);
	local= funA(k)+funB(k);
	printf("Dentro devuelve=%d ",local);
	
	return local;
}
